HTML

JavaScript API
// Get current weight value const weight = getWeightValue('topic-weight'); // Returns 1-10 // Set weight programmatically setWeightValue('topic-weight', 8); // Handle weight changes document.getElementById('topic-weight-slider') .addEventListener('change', () => { console.log('Weight changed to:', getWeightValue('topic-weight')); });
